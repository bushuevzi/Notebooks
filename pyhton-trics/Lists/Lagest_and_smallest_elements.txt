====== Lagest and smallest elements ======
Created четверг 04 Август 2016

import heapq
nums = [1, 8, 2, 23, 7, -4, 18, 23, 42, 37, 2]
print(heapq.nlargest(3, nums)) # Prints [42, 37, 23]
print(heapq.nsmallest(3, nums)) # Prints [-4, 1, 2]

In [15]: max(nums)
Out[15]: 42

In [21]: min(nums)
Out[21]: -4
